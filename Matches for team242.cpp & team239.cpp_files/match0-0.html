#include<iostream>
using namespace std;

int C(int a, int b)
{
    int ans = 1;
    for (int i = a; i > a - b; i--) ans *= i;
    for (int i = b; i > 1; i--) ans /= i;
    return ans;
}


int main() {
    int num = 0;
    int lca = 1;
    cin >> num;
    if (num == 1 || num == 0)
    {
        cout << 0 << endl << 0 << endl;
    }
    else
    {
        cout << C(num - 1, 2) * 2 << endl;
        cout << lca;
    }
    return 0;
}
