#include<iostream>
using namespace std;


int cao(int a, int b)
{
	int que = 1;
	for (int i = a; i > a - b; i--) que *= i;
	for (int i = b; i > 1; i--) que /= i;
	return que;
}


int  main() {
	int n = 0;
	int p = 1;
	cin >> n;
	if (n == 1 )
	{
		cout << 0 << endl << 0 << endl;
	}
	else
	{
		cout << cao(n - 1, 2) * 2 << endl;
		cout << p;
	}
	system("pause");
	return 0;
}
