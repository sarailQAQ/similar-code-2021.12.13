import java.util.*;
public class Main {
    public static void main(String args[]) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int t = sc.nextInt();

        for(int i = 0; i < t; i++) {
            int num = sc.nextInt();
            long carry = 1;
            int ans = 0;
            for (int k = 0; n >= carry; ++k) {
                ans += (n / (carry * 10)) * carry;
                ans +=  Math.min(Math.max(n % (carry * 10) - num * carry + 1, 0), carry);
                carry *= 10;
            }
            System.out.println(ans);
        }
    }
}
